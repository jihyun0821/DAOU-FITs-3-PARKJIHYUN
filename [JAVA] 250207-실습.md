```java
package assignment.day4.javaexam2;

import java.util.Arrays;
import java.util.StringTokenizer;

public class Test03 {
    public static void main(String[] args) {
        String str = "1.22,4.12,5.93,8.71,9.34";
        double[] data = new double[5];
        double sum = 0;

        StringTokenizer st = new StringTokenizer(str, ",");
        for(int i = 0; st.hasMoreTokens(); i++) {
            data[i] = Double.parseDouble(st.nextToken());
        }

        for(double x : data) {
            sum += x;
        }
        System.out.printf("합계 : %.3f\n", sum);
        System.out.printf("평균 : %.3f", sum/data.length);
    }
}
```

# Plane

### `Plane.java`

```java
package assignment.day4.javaexam2.plane;

public abstract class Plane {
    private String planeName;
    private int fuelSize;

    public Plane() {}

    public Plane(String planeName, int fuelSize) {
        this.planeName = planeName;
        this.fuelSize = fuelSize;
    }

    public void refuel(int fuel) {
        this.fuelSize += fuel;
    }

    @Override
    public String toString() {
        return this.planeName + "          " +
                this.fuelSize;
    }

    public abstract void flight(int distance);

    public String getPlaneName() {
        return planeName;
    }

    public void setPlaneName(String planeName) {
        this.planeName = planeName;
    }

    public int getFuelSize() {
        return fuelSize;
    }

    public void setFuelSize(int fuelSize) {
        this.fuelSize = fuelSize;
    }
}
```

### `Employee.java`

```java
package assignment.day4.javaexam2.company;

public abstract class Employee {
    private String name;
    private int number;
    private String department;
    private int salary;

    public Employee() {}
    public Employee(String name, int number, String department, int salary) {
        this.name = name;
        this.number = number;
        this.department = department;
        this.salary = salary;
    }

    public abstract double tax();

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getNumber() {
        return number;
    }

    public void setNumber(int number) {
        this.number = number;
    }

    public String getDepartment() {
        return department;
    }

    public void setDepartment(String department) {
        this.department = department;
    }

    public int getSalary() {
        return salary;
    }

    public void setSalary(int salary) {
        this.salary = salary;
    }

    @Override
    public String toString() {
        return name + "     " + department + "      " + salary;
    }
}
```

### `Company.java`

```java
package assignment.day4.javaexam2.company;

import java.util.*;

public class Company {
    public static void main(String[] args) {
        Map<Integer, Employee> map = new HashMap<>();

        // 1번의 데이터를 기반으로 객체를 생성 하여 HashMap에 넣는다.
        // HashMap에 넣을 때 키 값은 각 객체의 Number로 한다.
        map.put(1, new Secretary("Hilery", 1, "secretary", 800));
        map.put(2, new Sales("Clinten", 2, "sales", 1200));

        // 모든 객체의 기본 정보를 출력 한다(for문을 이용하여 출력 한다.)
        for (String s : Arrays.asList("name     department     salary ", "------------------------------------")) {
            System.out.println(s);
        }
        for(Employee employee : map.values()) {
            System.out.println(employee);
        }

        // 모든 객체에 인센티브 100씩 지급하고 다시 HashMap에 넣는다.
        System.out.println("\n인센티브 100 지급");
        for(Employee employee : map.values()) {
            if(employee instanceof Secretary) {
                ((Secretary) employee).incentive(100);
            }
            else if(employee instanceof Sales) {
                ((Sales) employee).incentive(100);
            }
        }

        // 모든 객체의 정보와 세금을 출력 한다 (for문을 이용하여 출력 한다.)
        printEmployee(map);

    }

    static void printEmployee(Map map) {
        System.out.println("name     department     salary     tax");
        System.out.println("----------------------------------------");

        for(Object employee : map.values()) {
            if(employee instanceof Secretary) {
                System.out.println(employee + "   " + ((Secretary) employee).tax());
            }
            else if(employee instanceof Sales) {
                System.out.println(employee + "   " + ((Sales) employee).tax());
            }
        }
    }
}
```
